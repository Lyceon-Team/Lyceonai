Goal

Step 3.1: Decide which table is authoritative for stats

You already use answer_attempts (session-scoped). Use that for session stats.

Implement helper:

async function getSessionStats(sessionId: string, userId: string) {
  const { data, error } = await supabaseServer
    .from("answer_attempts")
    .select("is_correct, outcome, attempted_at")
    .eq("session_id", sessionId)
    .eq("user_id", userId)
    .order("attempted_at", { ascending: false });

  if (error) return { correct: 0, total: 0, streak: 0 };

  const attempts = data ?? [];
  const correct = attempts.filter(a => a.is_correct === true).length;
  const total = attempts.length;

  let streak = 0;
  for (const a of attempts) {
    if (a.outcome === "skipped") continue;
    if (a.is_correct) streak++;
    else break;
  }

  return { correct, total, streak };
}
