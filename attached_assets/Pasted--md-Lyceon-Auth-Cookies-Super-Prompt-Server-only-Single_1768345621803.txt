## ✅ CHANGE 3 — One cookie-clearing implementation only (server-only)

We will consolidate cookie clearing into ONE module and delete the duplicate Google helper.

### 3A) Create new file: `server/lib/auth-cookies.ts`

Create this file with EXACT contents:

```ts
import type { Response } from "express";

export function clearAuthCookies(res: Response | any, isProd: boolean) {
  // Clear cookie variants deterministically:
  // - both tokens
  // - both paths historically used ('/' and '/api')
  // - both domain-scoped and host-only variants
  // - both secure=true/false (to clear legacy mis-scoped cookies)
  const names = ['sb-access-token', 'sb-refresh-token'] as const;
  const paths = ['/', '/api'] as const;

  const baseCommon = {
    httpOnly: true,
    sameSite: 'lax' as const,
  };

  const optionSets: any[] = [];

  for (const path of paths) {
    // Host-only clears (no domain)
    optionSets.push({ ...baseCommon, path, secure: false });
    optionSets.push({ ...baseCommon, path, secure: true });

    // Production domain clears (both dot-domain and non-dot legacy)
    if (isProd) {
      optionSets.push({ ...baseCommon, path, secure: true, domain: '.lyceon.ai' });
      optionSets.push({ ...baseCommon, path, secure: true, domain: 'lyceon.ai' });
      // In case a legacy cookie was set insecure in prod (rare but possible)
      optionSets.push({ ...baseCommon, path, secure: false, domain: '.lyceon.ai' });
      optionSets.push({ ...baseCommon, path, secure: false, domain: 'lyceon.ai' });
    }
  }

  for (const name of names) {
    for (const opts of optionSets) {
      res.clearCookie(name, opts);
    }
  }
}
```
