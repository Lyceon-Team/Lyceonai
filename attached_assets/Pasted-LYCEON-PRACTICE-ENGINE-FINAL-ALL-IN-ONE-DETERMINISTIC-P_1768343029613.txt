PART C1 — Mastery table schema (NEW)

Create a migration or SQL Editor script:

create table if not exists public.user_competencies (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users(id) on delete cascade,
  section text not null check (section in ('Math','Reading')),
  domain text null,
  skill text null,
  subskill text null,
  structure_cluster_id uuid null,

  attempts integer not null default 0,
  correct integer not null default 0,
  incorrect integer not null default 0,

  mastery_score numeric not null default 0, -- 0.0 → 1.0
  updated_at timestamptz not null default now(),

  unique (
    user_id,
    section,
    domain,
    skill,
    subskill,
    structure_cluster_id
  )
);

alter table public.user_competencies enable row level security;

create policy user_competencies_self
on public.user_competencies
for all
to authenticated
using (user_id = auth.uid())
with check (user_id = auth.uid());
