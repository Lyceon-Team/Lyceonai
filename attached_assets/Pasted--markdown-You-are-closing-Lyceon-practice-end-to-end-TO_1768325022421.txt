````markdown
Questions repeat because "Skip" does NOT record an attempt.

```ts
const handleSkip = async () => {
  if (!currentQuestion || !sessionId) return;

  setIsValidating(true);
  try {
    await apiRequest(`/api/practice/answer`, {
      method: "POST",
      body: JSON.stringify({
        sessionId,
        questionId: currentQuestion.id,
        skipped: true,
        selectedAnswer: null,
        freeResponseAnswer: null,
        elapsedMs: 0,
      }),
    });

    setScore((prev) => ({
      ...prev,
      skipped: prev.skipped + 1,
      total: prev.total + 1,
      streak: 0,
    }));
  } catch (e) {
    console.error("Error skipping question:", e);
    // Still allow advance, but repetition risk remains if skip isn't recorded
  } finally {
    setIsValidating(false);
    await handleNext(); // uses the existing next flow
  }
};
````
