```ts
function clearAuthCookies(res: any, isProd: boolean) {
  // Clear cookie variants deterministically:
  // - both tokens
  // - both paths historically used ('/' and '/api')
  // - both domain-scoped and host-only variants
  // - both secure=true/false (to clear legacy mis-scoped cookies)
  const names = ['sb-access-token', 'sb-refresh-token'] as const;
  const paths = ['/', '/api'] as const;

  const baseCommon = {
    httpOnly: true,
    sameSite: 'lax' as const,
  };

  const optionSets: any[] = [];

  for (const path of paths) {
    // Host-only clears (no domain)
    optionSets.push({ ...baseCommon, path, secure: false });
    optionSets.push({ ...baseCommon, path, secure: true });

    // Production domain clears (both dot-domain and non-dot legacy)
    if (isProd) {
      optionSets.push({ ...baseCommon, path, secure: true, domain: '.lyceon.ai' });
      optionSets.push({ ...baseCommon, path, secure: true, domain: 'lyceon.ai' });
      // In case a legacy cookie was set insecure in prod (rare but possible)
      optionSets.push({ ...baseCommon, path, secure: false, domain: '.lyceon.ai' });
      optionSets.push({ ...baseCommon, path, secure: false, domain: 'lyceon.ai' });
    }
  }

  for (const name of names) {
    for (const opts of optionSets) {
      res.clearCookie(name, opts);
    }
  }
}
```

✅ This is the core “stale cookie” fix. It prevents split-brain behavior caused by old `/api`-scoped or host-only cookie variants.

```ts
function requireAdminOrBypass(req: Request, res: Response, next: NextFunction): void {
  // SINGLE SOURCE OF TRUTH:
  // server/index.ts mounts this router behind v4AdminMiddleware which either:
  // - sets __v4Bypassed for allowlisted debug endpoints, OR
  // - enforces requireSupabaseAdmin (admin-only).
  //
  // Therefore, this middleware must NOT use apps/api auth logic.
  // It may only do a lightweight defense-in-depth check using req.user (set by server middleware).

  if ((req as any).__v4Bypassed) {
    next();
    return;
  }

  const role = (req as any).user?.role;
  if (role === 'admin') {
    next();
    return;
  }

  return res.status(401).json({ ok: false, error: "Admin access required" });
}
```
